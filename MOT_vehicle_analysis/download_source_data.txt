create table mot.test_data  as
select 
       test_id          
      ,vehicle_id       
      ,test_date        
      ,test_class_id    
      ,test_type        
      ,test_result      
      ,test_mileage     
      ,postcode_area    
      ,make             
      ,model            
      ,colour           
      ,fuel_type        
      ,cylinder_capacity
      ,first_use_date       
from  read_csv ('pipe*'
                , header=True
                ,quote='^'
                ,DATEFORMAT = '%Y-%m-%d'
                ,ESCAPE = '\'
                ,ignore_errors = True
                ,sep='|'
                ,columns = 
                {
                 'row_num'           : 'BIGINT'
                ,'test_id'           : 'BIGINT'
                ,'vehicle_id'        : 'BIGINT'
                ,'test_date'         : 'DATE'
                ,'test_class_id'     : 'BIGINT'
                ,'test_type'         : 'VARCHAR'
                ,'test_result'       : 'VARCHAR'
                ,'test_mileage'      : 'BIGINT'
                ,'postcode_area'     : 'VARCHAR'
                ,'make'              : 'VARCHAR'
                ,'model'             : 'VARCHAR'
                ,'colour'            : 'VARCHAR'
                ,'fuel_type'         : 'VARCHAR'
                ,'cylinder_capacity' : 'BIGINT'
                ,'first_use_date'    : 'DATE'               
                }
               )
;
